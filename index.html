C seat_allotment.c 
// Seat Booking System using Heap-based structure

#define MAX_SEATS 20

typedef struct {
    char name[50], phone[10];
    char source[50], destination[50];
    int seat_number;
    char reserved_for; // 'M', 'F', or 'N'
} Customer;

typedef struct {
    Customer customers[MAX_SEATS];
    bool is_booked[MAX_SEATS];
    int capacity;
} SeatHeap;

int book_seat(SeatHeap* heap, char gender, char* name,
              char* phone, char* source, char* destination) {
    for (int i = 0; i < heap->capacity; i++) {
        if (!heap->is_booked[i] &&
           (heap->customers[i].reserved_for == gender ||
            heap->customers[i].reserved_for == 'N')) {
            heap->is_booked[i] = true;
            heap->customers[i].seat_number = i + 1;
            strcpy(heap->customers[i].name, name);
            printf("Seat %d booked for %s.\n", i+1, name);
            return i + 1;
        }
    }
    return -1; // No available seats
}

C++ trash_tsp.cpp 
// Greedy TSP for Trash Route Optimization

const int INF = 1e9;

int findNearest(int current,
                const vector<vector<int>>& graph,
                vector<bool>& visited) {
    int n = graph.size();
    int nearest = -1, minDist = INF;
    for (int i = 0; i < n; ++i) {
        if (!visited[i] && graph[current][i] < minDist) {
            minDist = graph[current][i];
            nearest = i;
        }
    }
    return nearest;
}

void greedyTSP(const vector<vector<int>>& graph) {
    int n = graph.size();
    vector<bool> visited(n, false);
    visited[0] = true;
    int current = 0, totalDist = 0;

    cout << "Route: 0 ";
    for (int i = 1; i < n; ++i) {
        int next = findNearest(current, graph, visited);
        if (next == -1) break;
        totalDist += graph[current][next];
        current = next;
        visited[current] = true;
        cout << "→ " << current << " ";
    }
    totalDist += graph[current][0];
    cout << "→ 0\nTotal Distance: " << totalDist << " units\n";
}
